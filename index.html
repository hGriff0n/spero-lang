# This code is editable and runnable! mod spero:demo def main = () -> { # A simple integer calculator: # `+` or `-` means add or subtract by 1 # `*` or `/` means multiply or divide by 2 let program = "+ + * - /"; let acc = mut 0; for char in program do char.match { case '+' -> acc += 1 case '-' -> acc -= 1 case '*' -> acc *= 2 case '/' -> acc /= 2 case _ -> () } "The program \"{}\" evaluates to {}" .std:io:println(program, acc) }
